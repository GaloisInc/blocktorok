model StructureModel (qw) {
  technique : FEM;
  boundaryField : (Everywhere,  Neumann, qw);
  physics : HeatTransfer{P};
  const rhos = P.SolidDensity;
  const ks = P.SolidThermalConductivity;
  const cps = P.SolidSpecificHeat;
  var Tw;
  rhos*cps*∇Tw- ks*△Tw = qw;
  solve Tw with {?, ?};
  return Tw;
  }
